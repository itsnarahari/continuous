JUnit is a testing framework used in Spring Boot to write unit and integration tests.

import static org.junit.jupiter.api.Assertions.*;
import org.junit.jupiter.api.Test;

class EmployeeServiceTest {
    @Test
    void testGetEmployeeNameById() {
        assertEquals("John Doe", new EmployeeService().getEmployeeNameById(1));
    }
}
